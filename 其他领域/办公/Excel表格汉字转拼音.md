Function ChangePY(str)
For i = 1 To Len(str)
    PY = PinYin(Mid(str, i, 1))
    PY = UCase(Left(PY, 1)) & Right(PY, Len(PY) - 1)
    ChangePY = ChangePY & PY
Next i
End Function
